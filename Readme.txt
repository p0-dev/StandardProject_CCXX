# Building Manual Using CMake / Ninja / User's Toolchain
; =======================================================================================
; BUILD SEQUENCE OVERVIEW
; =======================================================================================
; STEP_0       - PREPARATION
; Description  : Removing previous results file inside <Build>, <Diagram>, <Install>
; Input        : N/A
; Output       : Clean directories <Build>, <Diagram>, <Install>

; STEP_1       - GENERATION
; Description  : Using CMake for generating Makefile
; Input        : CMakeLists.txt
; Output       : Makefile / Ninja build instruction file

; STEP_2       - BUILDING
; Description  : Using build tool/compiler to build the project
; Input        : Makefile / Ninja build instruction file
; Output       : Library/execution file, depends on target

; STEP_3       - INSTALLATION
; Description  : Using Cmake for installing target to system
; Input        : CMake's install script from STEP 1
; Output       : Header / library / execution are being copied to the system


; =======================================================================================
; GENERATION PROCEDURE
; =======================================================================================

; =======================================================================================
; CMAKE FLAGS
; =======================================================================================
; Project Directory                    | -S . (at current directory) 
;                                      | (required)
; ---------------------------------------------------------------------------------------
; Build Directory                      | -B Build (results will be at <Build>) 
;                                      | (required)
; ---------------------------------------------------------------------------------------
; Build Tool                           | -G Ninja (default build tool) 
;                                      | (required)
; ---------------------------------------------------------------------------------------
; Toolchain                            | --toolchain Toolchain/Mingw64.cmake 
;                                      | (required) 
;                                      | (default toolchain)
; ---------------------------------------------------------------------------------------
; Developer Warning                    | -Wdev 
;                                      | (optional)
; ---------------------------------------------------------------------------------------
; Developer Warning                    | -Werror=dev 
;                                      | (optional)
; ---------------------------------------------------------------------------------------
; Cleaning                             | --fresh
;                                      | (optional)
; ---------------------------------------------------------------------------------------
; Automatic Diagram                    | --graphviz=Diagram/Overview.dot 
; ---------------------------------------------------------------------------------------
; Debug Option                         | --check-system-vars
; ---------------------------------------------------------------------------------------
; Compile Type Option                  | -D CMAKE_BUILD_TYPE=Debug
;                                      | Available build types:
;                                      | - Debug
;                                      | - Release
; =======================================================================================

## Example
cmake   -S . \
        -B Build \
        -G Ninja \
        --toolchain Toolchain/Mingw64.cmake \
        -Wdev \
        -Werror=dev \
        --fresh \
        --graphviz=Diagram/Overview.dot \
        --check-system-vars \
        -D CMAKE_BUILD_TYPE=Debug


; =======================================================================================
; BUILDING PROCEDURE
; =======================================================================================

; ---------------------------------------------------------------------------------------
; CMAKE FLAGS
; ---------------------------------------------------------------------------------------
; Building Flag    | --build Build (specify build directory)
;                  | Example: directory <Build>
;                  | (required)
; ---------------------------------------------------------------------------------------
; Performance Flag | --parallel
;                  | (optional)
; ---------------------------------------------------------------------------------------
; Config Flag      | --config Debug (specify Debug / Release build)
;                  | Available options:
;                  | - Debug
;                  | - Release
;                  | (required)
; ---------------------------------------------------------------------------------------
; Cleaning Flag    | --clean-first
;                  | (required)
; ---------------------------------------------------------------------------------------
; Debug Flag       | --verbose
;                  | (optional)
; ---------------------------------------------------------------------------------------

## Example
cmake   --build Build \
        --parallel \
        --config Debug \
        --clean-first \
        --verbose \


; =======================================================================================
; INSTALLATION PROCEDURE
; =======================================================================================

; ---------------------------------------------------------------------------------------
; CMAKE FLAGS
; ---------------------------------------------------------------------------------------
; Install Flag     | --install Build (specify installation instruction directory)
;                  | Example: directory <Build>
;                  | (required)
; ---------------------------------------------------------------------------------------
; Install Flag     | --prefix Install (specify result directory)
;                  | Example: directory <Install>
;                  | (required)
; ---------------------------------------------------------------------------------------
; Cleaning Flag    | --strip
;                  | (required)
; ---------------------------------------------------------------------------------------
; Debug Flag       | --verbose
;                  | (optional)
; ---------------------------------------------------------------------------------------

## Example
cmake   --install Build \
        --prefix Install \
        --strip \
        --verbose
